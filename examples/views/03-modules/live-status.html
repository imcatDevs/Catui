const { OnlineStatus } = await IMCAT.use('live-status');
const os = new OnlineStatus('#avatar', {
  status: 'online',  // 'online' | 'offline' | 'away' | 'busy' | 'dnd'
  size: 'md',       // 'sm' | 'md' | 'lg'
  pulse: true,     // 온라인 시 펄스 애니메이션
  position: 'bottom-right'
});
os.setStatus('away'); // 상태 변경
TypingIndicator

채팅에서 상대방이 입력 중임을 표시. 사용자 추가/제거 API 지원.

const { TypingIndicator } = await IMCAT.use('live-status');
const ti = new TypingIndicator('#container', {
  users: ['홍길동'],
  maxDisplay: 3,
  showNames: true
});
ti.addUser('김영희');
ti.removeUser('홍길동');
ti.clear();
LiveCounter

실시간으로 변하는 숫자 표시. 애니메이션 전환, 천 단위 구분자, 접두/접미사 지원.

접속자 수

매출

완료율

const { LiveCounter } = await IMCAT.use('live-status');
const c = new LiveCounter('#container', {
  value: 1234,
  prefix: '',
  suffix: ' 명',
  separator: ',',
  duration: 1000
});
c.setValue(5678);    // 애니메이션 전환
c.increment(100);  // +100
c.decrement(50);   // -50
ConnectionStatus

네트워크 연결 상태 자동 감지. 오프라인 시 상단 배너 표시, 온라인 복귀 시 자동 숨김. 싱글톤 — selector 없이 생성.

네트워크를 끊으면 상단에 빨간 배너가 표시됩니다. 아래 버튼으로 시뮬레이션할 수 있습니다.

const { ConnectionStatus } = await IMCAT.use('live-status');
new ConnectionStatus({
  showOnline: true,
  position: 'top',         // 'top' | 'bottom'
  autoHide: true,
  autoHideDelay: 3000,
  onStatusChange: (isOnline) => {
    console.log(isOnline ? '온라인' : '오프라인');
  }
});
// 싱글톤: 두 번째 생성 시 기존 인스턴스 반환
ActivityStatus

마지막 활동 시간 표시. 상대 시간(방금 전, 5분 전 등) 또는 절대 시간 형식. 자동 갱신.

방금 활동

30분 전

3시간 전

2일 전

const { ActivityStatus } = await IMCAT.use('live-status');
const as = new ActivityStatus('#container', {
  lastActivity: new Date(),
  format: 'relative',    // 'relative' | 'absolute'
  updateInterval: 60000, // 1분마다 갱신
  prefix: '마지막 활동: '
});
as.updateNow(); // 현재 시간으로 갱신

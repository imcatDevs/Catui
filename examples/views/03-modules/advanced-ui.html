← 수평 패널 크기 변경 
const { SplitPane } = await IMCAT.use('advanced-ui');

// 수평 분할
const split = new SplitPane('#container', {
  direction: 'horizontal',
  initialSizes: [50, 50],
  minSizes: [80, 80],
  gutterSize: 8,
  collapsible: true,
  onDragEnd: (sizes) => console.log(sizes)
});

// 프로그래밍 방식 크기 변경
split.setSizes([30, 70]);
QRCode

QR Server API 기반 QR 코드 생성 — 실제 스캔 가능, 다운로드 지원

const { QRCode } = await IMCAT.use('advanced-ui');

const qr = new QRCode('#container', {
  text: 'https://imcat.dev',
  size: 200,
  colorDark: '000000',
  correctLevel: 'M'
});

// 텍스트 변경
qr.setText('새로운 URL');

// 다운로드
qr.download('my-qr.png');

CopyToClipboard

버튼 바인딩 방식 + static 유틸리티 — 자동 피드백 포함

인스턴스 방식 (버튼 바인딩)

Static 메서드 방식

복사됨! 
const { CopyToClipboard } = await IMCAT.use('advanced-ui');

// 1. 인스턴스 방식 — 버튼에 바인딩
new CopyToClipboard('#button', {
  target: '#input',
  feedbackText: '복사됨!',
  onSuccess: (text) => console.log('복사:', text)
});

// 2. Static 방식 — 직접 호출
const ok = await CopyToClipboard.copy('텍스트');
const ok2 = await CopyToClipboard.copyFrom('#element');
SimpleColorPicker

프리셋 색상 선택기 — native color input 연동, 커스텀 팔레트

const { SimpleColorPicker } = await IMCAT.use('advanced-ui');

new SimpleColorPicker('#container', {
  colors: ['#667eea', '#f56565', '#48bb78'],
  value: '#3b82f6',
  showInput: true,
  onChange: (color) => console.log(color)
});

// 프로그래밍 방식
picker.setValue('#ef4444');
picker.getValue();  // '#ef4444'

Navigation

Sidebar, Tabs, Accordion, Collapse, MegaMenu, TreeView 컴포넌트

Sidebar

다단계 메뉴, 뱃지, 토글(expanded/collapsed) 지원.

사이드바 메뉴 항목을 클릭하면 여기에 표시됩니다.
const { Sidebar } = await IMCAT.use('navigation');
const sidebar = new Sidebar('#container', {
  mode: 'expanded',       // expanded | collapsed | compact
  showUserBox: true,
  user: { name: '홍길동', role: 'Admin' },
  activeItem: 'dashboard',
  items: [
    { type: 'title', label: '메뉴' },
    { id: 'dashboard', label: '대시보드', icon: 'dashboard',
      badge: { text: '4', variant: 'success' } },
    { label: '앱', icon: 'apps', children: [
      { label: '캘린더', href: '#' },
      { label: '채팅', href: '#' }
    ]}
  ],
  onItemClick: function(item, link) { /* ... */ }
});
sidebar.setMode('collapsed'); // 프로그래밍 방식 토글

Accordion

Separated 스타일 (multiple: true)

const { Accordion } = await IMCAT.use('navigation');
// .accordion__item > .accordion__header + .accordion__content 구조 필요
new Accordion('#container .accordion', {
  multiple: false,     // true면 여러 개 동시 열기
  expandFirst: true,
  animation: true
});
Collapse
이 영역은 Collapse 컴포넌트로 접었다 펼 수 있습니다. 애니메이션이 포함됩니다.
const { Collapse } = await IMCAT.use('navigation');
const collapse = new Collapse('#collapse-demo', { expanded: false });
document.getElementById('btn').addEventListener('click', () => collapse.toggle());
TreeView
const { TreeView } = await IMCAT.use('navigation');
// .treeview__item > .treeview__toggle + .treeview__label + .treeview__children 구조
const tree = new TreeView('#tree-demo');
tree.expandAll();
tree.collapseAll();
MegaMenu
const { MegaMenu } = await IMCAT.use('navigation');
// .megamenu__item > .megamenu__trigger + .megamenu__panel 구조
new MegaMenu('#mega-demo', {
  trigger: 'click', // hover | click
  closeOnOutside: true
});

Colors

CSS 변수 기반 색상 시스템. 라이트/다크 테마를 자동 지원합니다.

시맨틱 색상
/* CSS 변수로 접근 */
var(--primary)      /* 주 색상 */
var(--success-color) /* 성공 */
var(--danger-color)  /* 위험 */
var(--warning-color) /* 경고 */
var(--info-color)    /* 정보 */
배경 & 텍스트

text-primary — 주요 텍스트 색상

text-secondary — 보조 텍스트 색상

text-disabled — 비활성 텍스트

/* 배경 */
var(--bg-primary)    /* 주 배경 */
var(--bg-secondary)  /* 보조 배경 */
var(--bg-tertiary)   /* 3차 배경 */

/* 텍스트 */
var(--text-primary)  /* 주 텍스트 */
var(--text-secondary)/* 보조 텍스트 */
var(--text-tertiary) /* 3차 텍스트 */

/* 테두리 */
var(--border-color)  /* 기본 테두리 */
테마 전환
<!-- HTML data-theme 속성으로 전환 -->
<html data-theme="light">
<html data-theme="dark">

// JS로 전환
document.documentElement.dataset.theme = 'dark';

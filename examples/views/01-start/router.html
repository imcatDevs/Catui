Router

Hash 기반 SPA 라우터. catui-href 속성으로 자동 네비게이션합니다.

기본 사용법
<!-- HTML 속성으로 라우팅 -->
<body catui-target="main-content">
  <a catui-href="views/home.html">홈</a>
  <a catui-href="views/about.html">소개</a>
  <div id="main-content"></div>
</body>
JS API
// 프로그래밍 방식 네비게이션
IMCAT.view.navigate('views/dashboard.html');

// 현재 경로 조회
const path = IMCAT.view.current();

// URL 파라미터 조회
// URL: #views/product.html?id=123
const params = IMCAT.view.params();
console.log(params.id); // '123'
라이프사이클 훅
// 뷰 로드 후 콜백
IMCAT.view.afterLoad((path) => {
  console.log('뷰 로드됨:', path);
});

// 인스턴스 자동 정리
const modal = new Modal({ ... });
IMCAT.view.registerInstance(modal);
// → 다른 뷰로 이동 시 modal.destroy() 자동 호출

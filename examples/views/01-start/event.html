Event Bus

글로벌 pub/sub 이벤트 버스와 컴포넌트용 EventEmitterMixin.

글로벌 EventBus
이벤트 대기중...
// 글로벌 이벤트 구독
IMCAT.on('user:login', (data) => {
  console.log('로그인:', data.name);
});

// 이벤트 발행
IMCAT.emit('user:login', { name: '홍길동' });

// 1회성 구독
IMCAT.once('app:ready', () => { ... });

// 구독 해제
IMCAT.off('user:login');
EventEmitterMixin
컴포넌트 내부 이벤트용 경량 믹스인. 글로벌 EventBus와 분리됩니다.
// 컴포넌트에 이벤트 기능 추가
class MyComponent {
  constructor() {
    // 방법 1: 객체에 직접 적용
    EventEmitterMixin.apply(this);

    // 방법 2: 별도 이벤트 버스 생성
    this.events = EventEmitterMixin.create();
  }
}

const comp = new MyComponent();
comp.on('change', (val) => console.log(val));
comp.emit('change', '새 값');
comp.clearEvents(); // 모든 이벤트 해제

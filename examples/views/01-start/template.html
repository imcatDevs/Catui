Template

간단한 Mustache 스타일 템플릿 엔진. XSS 자동 방어가 내장되어 있습니다.

기본 사용
// 템플릿 문자열 렌더링
const html = IMCAT.template.render(
  '<h3>{{name}}</h3><p>{{message}}</p>',
  { name: '홍길동', message: '안녕하세요!' }
);

// 변수는 자동으로 XSS 이스케이프됨
// {{name}} → Security.escape(name)
조건부 렌더링
보임
// {{#key}} ... {{/key}} — truthy 일 때 렌더링
// {{^key}} ... {{/key}} — falsy 일 때 렌더링
IMCAT.template.render(
  '{{#show}}보임{{/show}}{{^show}}숨김{{/show}}',
  { show: true }
);
반복 렌더링
    // 배열 반복 — {{.}} 은 현재 값
IMCAT.template.render(
  '{{#items}}<li>{{.}}</li>{{/items}}',
  { items: ['사과', '바나나', '딸기'] }
);

// 객체 배열 — 각 필드 접근 가능
IMCAT.template.render(
  '{{#users}}<p>{{name}} ({{role}})</p>{{/users}}',
  { users: [{ name: '홍길동', role: 'Admin' }] }
);
    XSS 자동 방어
    // {{value}} → 자동 XSS 이스케이프
IMCAT.template.render(
  '<p>{{userInput}}</p>',
  { userInput: '<script>alert("XSS")</script>' }
);
// 결과: &lt;script&gt;alert(&quot;XSS&quot;)&lt;/script&gt;

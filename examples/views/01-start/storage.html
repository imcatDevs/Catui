Storage

localStorage 래퍼. JSON 직렬화/역직렬화, 만료 시간, 네임스페이스를 지원합니다.

기본 사용
// 저장 (JSON 자동 직렬화)
IMCAT.storage.set('user', { name: '홍길동', age: 30 });

// 조회 (JSON 자동 역직렬화)
const user = IMCAT.storage.get('user');
// { name: '홍길동', age: 30 }

// 기본값 지정
const theme = IMCAT.storage.get('theme', 'light');

// 삭제
IMCAT.storage.remove('user');

// 전체 초기화
IMCAT.storage.clear();
JSON 객체 저장
저장된 데이터 없음
// 객체/배열도 자동 JSON 직렬화
IMCAT.storage.set('user', { name: '홍길동', age: 30, roles: ['admin'] });

// 조회 시 자동 역직렬화 (JSON.parse)
var user = IMCAT.storage.get('user');
console.log(user.name); // '홍길동'
저장소 관리
// 전체 초기화
IMCAT.storage.clear();
Helpers 단축 접근
// helpers 네임스페이스로도 접근 가능
IMCAT.helpers.setStorage('key', { data: 123 });
IMCAT.helpers.getStorage('key', defaultValue);
